package agenda_de_tarefas;

import java.util.Scanner;

public class AppTarefa {

	public static void main(String[] args) {
		Scanner scanner = new Scanner(System.in);
		
		System.out.println("----- MENU PRINCIPAL -----");
		System.out.println("1 - Inclusão de tarefa");
		System.out.println("2 - Consulta de tarefa");
		System.out.println("3 - Alteração de tarefa");
		System.out.println("4 - Exclusão de tarefa");
		System.out.println("0 - F I M");
		System.out.println("Digite a opção desejada:");
		while(true) {
			int opcao = scanner.nextInt();
			if(opcao == 1) {
			System.out.println("Opção escolhida 'Inclusão de tarefa' ");
			}
			else if(opcao == 2) {
			System.out.println("Opção escolhida 'Consulta de tarefa' ");
			}
			else if(opcao == 3) {
			System.out.println("Opção escolhida 'Alteração de tarefa' ");
			}
			else if(opcao == 4) {
			System.out.println("Opção escolhida 'Exclusão de tarefa' ");
			}
			else if(opcao == 0) {
			System.out.println("Encerrando aplicação");
			break;
			}
			else {
			System.out.println("Digite uma opção valida");
			}

		}

	}
}
